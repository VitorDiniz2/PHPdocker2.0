CREATE DATABASE IF NOT EXISTS eduverde;
USE eduverde;

CREATE TABLE IF NOT EXISTS user (
  id INT(11) AUTO_INCREMENT,
  nome VARCHAR(100),
  cpf INT (11),
  email VARCHAR(100),
  senha VARCHAR(50),
  foto VARCHAR(),
  PRIMARY KEY (id)
);
